Model evaluation¶

In [ ]:
from typing import NamedTuple

class EvaluationResult(NamedTuple):
    accuracy_score: float
    precision_score: float
    recall_score: float
    f1_score: float
    auc_score: float

    def __repr__(self) -> str:
        return "\n".join((f"{name}: {val:.2f}" for name, val in self._asdict().items()))
In [ ]:
from sklearn.metrics import confusion_matrix, accuracy_score, precision_score, recall_score, f1_score, \
    roc_auc_score, RocCurveDisplay
import seaborn as sns

def evaluate_model(Y_true, Y_pred, Y_pred_proba):
    # plot confusion matrix
    cf_matrix = confusion_matrix(Y_pred, Y_true, labels=[0,1])
    sns.heatmap(cf_matrix, annot=True)
    # plot ROC curve
    RocCurveDisplay.from_predictions(Y_true, Y_pred_proba[:, 1])

    return EvaluationResult(
        accuracy_score=accuracy_score(Y_pred, Y_true)*100,
        precision_score=precision_score(Y_pred, Y_true, pos_label=0)*100,
        recall_score=recall_score(Y_pred, Y_true, pos_label=0)*100,
        f1_score=f1_score(Y_pred, Y_true, pos_label=0)*100,
        auc_score=roc_auc_score(Y_true, Y_pred_proba[:, 1])*100
    )
